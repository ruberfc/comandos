#comprobar

#asignar email a proyecto
git config --global user.email "ruberfc@gmail.com"

#comprobar
git config --global user.name

#estado general del proyecto
git status

#agregar al state area
git add "comandos-git.txt" 贸 git add -A 贸 git add .

#hacer la fotografia de tiempo o commit
git commit -m "act 27-07-21 11:57"

#ver todos los commits
git log

#ver todos los commits mas resumidos
git log --oneline

#ver diferencias en el archivo
git diff

#quitar el archivo de state area
git restore --staged "comandos-git.txt"

#volver a un punto de tiempo especifico
git checkout eea7929

#regresar al punto de tiempo principal
git checkout master

#ver la diferencia entre los commits especificos
git diff eea7929 d3a905a

#restaurar fragmentos de archivos antes del agregar al state area
git restore comandos-git.txt 贸 git restore .

#ver la ramas del proyecto
git branch

#crear la rama esclavo
git branch esclavo

#moverse a la rama esclavo
git checkout esclavo 贸 git switch esclavo

#juntar los cambios de esclavo a la rama principal
git merge esclavo

#ver la union de los cambios
git log --oneline --graph
